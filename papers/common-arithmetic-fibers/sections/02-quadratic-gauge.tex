\section{The universal quadratic gauge}\label{sec:gauge}

Let
\begin{equation}
 G(S)=g_1S+g_2S^2+\cdots+g_NS^N\in K[S],
 \qquad g_1g_3g_N\neq0.
 \label{eq:seed}
\end{equation}
In source coordinates $(x,y,z)$ put
\begin{equation}
 t=1+xy,
 \qquad
 q=t^2z+\frac{g_1}{g_3}y^2(1+3t).
 \label{eq:tq}
\end{equation}
Define $F_G=(\Pi,B,C):\A^3_K\to\A^3_K$ by
\begin{align}
 \Pi={}&tq,\notag\\
 B={}&y+3\frac{g_3}{g_1}xq+2\frac{g_2}{g_1}tq
       +\sum_{k=4}^N k\frac{g_k}{g_1}t^2x^{k-2}q^k,\notag\\
 C={}&x(5-3t)-\frac{g_3}{g_1}x^3z
       -\sum_{k=4}^N(k-2)\frac{g_k}{g_1}(xq)^k.
 \label{eq:gauge-map}
\end{align}
The formula is polynomial and uses only the coefficients of $G$.

For a target $(\Pi,B,C)$ define
\begin{equation}
 G_\Pi(S)=g_1S+\Pi(g_2S^2+g_3S^3)
          +\sum_{k=4}^Ng_k\Pi^kS^k
 \label{eq:Gpi}
\end{equation}
and
\begin{equation}
 E_{\Pi,B,C}(S)=G_\Pi(S)-\frac{g_1}{2}(BS^2+C).
 \label{eq:inverse}
\end{equation}

\begin{proposition}[Quadratic-gauge fiber theorem]\label{prop:gauge}
The map $F_G$ satisfies
\[
 \det DF_G=-2.
\]
Its generic inverse equation is $E_{\Pi,B,C}(S)=0$.  On the chart
$\Pi\neq0$, every simple root of this equation reconstructs one and only one
source point.  In particular, if $E_{\Pi,B,C}$ is squarefree of degree $N$,
then
\[
 F_G^{-1}(\Pi,B,C)\simeq\Spec K[S]/(E_{\Pi,B,C})
\]
and the fiber is a Keller fiber of rank $N$.
\end{proposition}

\begin{proof}
On $t\neq0$ introduce
\[
 S=\frac{x}{t},
 \qquad Q=y+xq,
 \qquad D=1-SQ+\Pi S^2=t^{-1}.
\]
The change $(x,y,z)\mapsto(\Pi,S,Q)$ has Jacobian $D^{-1}$.  A direct
coefficient calculation from \eqref{eq:gauge-map} gives
\[
 \frac{\partial E_{\Pi,B,C}}{\partial S}=g_1D
\]
on the source incidence.  Equivalently, at fixed $\Pi$ the marked-line map
$(S,Q)\mapsto(B,C)$ has Jacobian $-2D$.  The two factors cancel:
\[
 \det\frac{\partial(\Pi,B,C)}{\partial(x,y,z)}
 =(-2D)D^{-1}=-2.
\]

Conversely, given a root $S$ of \eqref{eq:inverse}, put
\[
 \beta(\Pi,S)
 =\frac{G_\Pi'(S)/g_1-1-\Pi S^2}{S}.
\]
The numerator is divisible by $S$, so $\beta$ is polynomial.  Then
\[
 Q=B-\beta(\Pi,S),
 \qquad D=\frac{G_\Pi'(S)}{g_1}-BS.
\]
For a simple root, $D\neq0$, and the remaining source coordinates are
uniquely recovered by
\[
 t=D^{-1},\qquad x=\frac{S}{D},\qquad
 y=Q-\Pi S,
\]
\[
 q=\Pi D,
 \qquad
 z=\frac{q-(g_1/g_3)y^2(1+3t)}{t^2}.
\]
If the inverse polynomial is squarefree of degree $N$, its $N$ geometric
roots therefore give the complete fiber.  The same equation has generic
degree $N$, proving $\gdeg(F_G)=N$.
\end{proof}
