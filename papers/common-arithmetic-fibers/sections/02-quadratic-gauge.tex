\section{The universal quadratic gauge}\label{sec:gauge}

Let
\begin{equation}
 G(S)=g_1S+g_2S^2+\cdots+g_NS^N\in K[S],
 \qquad g_1g_3g_N\neq0.
 \label{eq:seed}
\end{equation}
In source coordinates $(x,y,z)$ put
\begin{equation}
 t=1+xy,
 \qquad
 q=t^2z+\frac{g_1}{g_3}y^2(1+3t).
 \label{eq:tq}
\end{equation}
Define $F_G=(\Pi,B,C):\A^3_K\to\A^3_K$ by
\begin{align}
 \Pi={}&tq,\notag\\
 B={}&y+3\frac{g_3}{g_1}xq+2\frac{g_2}{g_1}tq
       +\sum_{k=4}^N k\frac{g_k}{g_1}
          t^2x^{k-2}q^k,\notag\\
 C={}&x(5-3t)-\frac{g_3}{g_1}x^3z
       -\sum_{k=4}^N(k-2)\frac{g_k}{g_1}(xq)^k.
 \label{eq:gauge-map}
\end{align}
The formula is polynomial and uses only the coefficients of $G$.

For a target $(\Pi,B,C)$ define
\begin{equation}
 G_\Pi(S)=g_1S+\Pi(g_2S^2+g_3S^3)
          +\sum_{k=4}^Ng_k\Pi^kS^k
 \label{eq:Gpi}
\end{equation}
and
\begin{equation}
 E_{\Pi,B,C}(S)=G_\Pi(S)-\frac{g_1}{2}(BS^2+C).
 \label{eq:inverse}
\end{equation}

\begin{proposition}[Quadratic-gauge fiber theorem]\label{prop:gauge}
The map $F_G$ satisfies
\[
 \det DF_G=-2.
\]
Its generic inverse equation is $E_{\Pi,B,C}(S)=0$.  Let
$(\pi,b,c)\in K^*\times K^2$.  If $E_{\pi,b,c}$ is squarefree of degree
$N$, then scheme-theoretically
\[
 F_G^{-1}(\pi,b,c)\simeq\Spec K[S]/(E_{\pi,b,c}).
\]
In particular, this is a Keller fiber of rank $N$.
\end{proposition}

\begin{proof}
On $t\neq0$ introduce
\[
 S=\frac{x}{t},
 \qquad Q=y+xq,
 \qquad D=1-SQ+\Pi S^2=t^{-1}.
\]
A direct differentiation gives
\begin{equation}
 \det\frac{\partial(\Pi,S,Q)}{\partial(x,y,z)}=D^{-1}.
 \label{eq:source-chart-jacobian}
\end{equation}
Now set
\begin{equation}
 \beta(\Pi,S)
 =\frac{G_\Pi'(S)/g_1-1-\Pi S^2}{S}.
 \label{eq:beta}
\end{equation}
The numerator is divisible by $S$.  Substitution in
\eqref{eq:gauge-map} gives the two identities
\begin{equation}
 B=Q+\beta(\Pi,S),
 \qquad
 C=\frac{2G_\Pi(S)}{g_1}-BS^2.
 \label{eq:marked-line-identities}
\end{equation}
The second identity is exactly $E_{\Pi,B,C}(S)=0$, while differentiation
gives
\begin{equation}
 D=\frac{G_\Pi'(S)}{g_1}-BS
  =\frac{E_{\Pi,B,C}'(S)}{g_1}.
 \label{eq:D-derivative}
\end{equation}
At fixed $\Pi$, differentiating \eqref{eq:marked-line-identities} with
respect to $(S,Q)$ gives
\begin{equation}
 \det\frac{\partial(B,C)}{\partial(S,Q)}=-2D.
 \label{eq:plane-jacobian}
\end{equation}
Combining \eqref{eq:source-chart-jacobian} and
\eqref{eq:plane-jacobian} proves $\det DF_G=-2$ on the dense chart
$t\neq0$, hence polynomially everywhere.

It remains to prove the fiber statement on coordinate rings.  Fix
$(\pi,b,c)$ as in the proposition and abbreviate
\[
 E=E_{\pi,b,c},
 \qquad R=K[S]/(E),
 \qquad s=S\bmod E.
\]
Squarefreeness gives $\gcd(E,E')=1$.  Thus there are $U,V\in K[S]$ with
\[
 UE+VE'=1,
\]
so $E'(s)$ is a unit of $R$, with inverse $V(s)$.  In particular no
localization has been hidden in the reconstruction.  Put
\[
 d=\frac{E'(s)}{g_1},
 \qquad q_0=b-\beta(\pi,s),
\]
and define in $R$
\begin{equation}
 \begin{aligned}
  t_0&=d^{-1},& x_0&=s\,d^{-1},& y_0&=q_0-\pi s,\\
  q_1&=\pi d,&
  z_0&=d^2\left(q_1-\frac{g_1}{g_3}y_0^2(1+3t_0)\right).
 \end{aligned}
 \label{eq:ring-reconstruction}
\end{equation}
Equations \eqref{eq:marked-line-identities}--\eqref{eq:D-derivative}
show directly in $R$ that
\[
 1+x_0y_0=t_0,
 \qquad
 t_0^2z_0+\frac{g_1}{g_3}y_0^2(1+3t_0)=q_1,
\]
and that $F_G(x_0,y_0,z_0)=(\pi,b,c)$.  Hence
\eqref{eq:ring-reconstruction} defines a homomorphism from the coordinate
ring of the fiber to $R$.

Conversely, in the coordinate ring of the fiber one has
\[
 tq=\Pi=\pi\in K^*.
\]
Therefore both $t$ and $q$ are units on the entire fiber scheme, not merely
at its geometric points.  The elements
\[
 S=\frac{x}{t},
 \qquad Q=y+xq
\]
are consequently global there, and \eqref{eq:marked-line-identities} gives
$E(S)=0$.  This defines the reverse homomorphism $R$ to the fiber ring.
Under the two maps,
\[
 D=t^{-1},\qquad x=S/D,\qquad y=Q-\pi S,
 \qquad q=\pi D,
\]
and the defining equation for $q$ recovers $z$.  Both compositions are the
identity.  This proves the asserted scheme isomorphism.

Finally, the generic polynomial \eqref{eq:inverse} has degree $N$ and is
separable, and the same reconstruction works at every generic root.  Hence
$\gdeg(F_G)=N$.
\end{proof}

\begin{corollary}[Jacobian-one effective form]\label{cor:effective-gauge}
Let
\[
 L(\Pi,B,C)=\left(\Pi,-\frac{B}{2},C\right),
 \qquad \widetilde F_G=L\circ F_G.
\]
Then
\[
 \det D\widetilde F_G=1,
 \qquad \gdeg(\widetilde F_G)=N.
\]
For every $(\pi,b,c)\in K^*\times K^2$ for which $E_{\pi,b,c}$ is
squarefree of degree $N$,
\[
 \widetilde F_G^{-1}\left(\pi,-\frac b2,c\right)
 \simeq\Spec K[S]/(E_{\pi,b,c}).
\]
In particular, every target with $b=0$ is fixed by the normalization.
Moreover, in total degree,
\[
 \deg\Pi\leq7,
 \qquad \deg B\leq6N+2,
 \qquad \deg C\leq6N,
\]
and therefore
\[
 \boxed{\max_i\deg(\widetilde F_G)_i\leq6N+2.}
\]
\end{corollary}

\begin{proof}
The map $L$ is a linear automorphism of determinant $-1/2$.  The determinant,
geometric-degree, and fiber statements therefore follow from
Proposition~\ref{prop:gauge}.

For the degree estimate, $\deg t=2$ and $\deg q\leq5$, hence
$\deg\Pi\leq7$.  The summand of $B$ indexed by $k\geq4$ has degree at most
\[
 2\deg t+(k-2)+k\deg q\leq4+(k-2)+5k=6k+2,
\]
while the three terms preceding the sum have degrees at most $1$, $6$, and
$7$.  Thus $\deg B\leq6N+2$.  Similarly, the summand $(xq)^k$ in $C$ has
degree at most $6k$, while the preceding terms have degrees at most $3$ and
$4$, so $\deg C\leq6N$.  Linear output normalization does not change these
bounds.
\end{proof}
